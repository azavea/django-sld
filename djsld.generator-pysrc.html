247      Accept a queryset of objects, and return the values of the class breaks

249      computed.

251      @type  classification: pysal classifier

255            - Equal_Interval 
256            - Fisher_Jenks 
257            - Jenks_Caspall 
258            - Jenks_Caspall_Forced

260            - Max_P_Classifier

262            - Natural_Breaks 
263            - Quantiles

269      @type  nclasses: integer 
270      @param nclasses: The number of class breaks desired. 
271      @type  geofield: string 
272      @param geofield: The name of the geography column on the model. Defaults to 'geom' 
273      @type  propertyname: string 
274      @param propertyname: The name of the filter property name, if different from the model field. 
275      @type  userstyletitle: string 
276      @param userstyletitle: The title of the UserStyle element. 
277      @type  featuretypestylename: string 
278      @param featuretypestylename: The name of the FeatureTypeStyle element. 
279      @type    kwargs: keywords 
280      @param   kwargs: Additional keyword arguments for the classifier. 
281      @rtype: L{sld.StyledLayerDescriptor} 
282      @returns: An SLD class object that represents the classification scheme  
283          and filters. 
284      """

286   
287      ftype = queryset.model._meta.get_field_by_name(geofield)[0] 
288      if isinstance(ftype, fields.LineStringField) or isinstance(ftype, fields.MultiLineStringField): 
289          symbolizer = LineSymbolizer 
290      elif isinstance(ftype, fields.PolygonField) or isinstance(ftype, fields.MultiPolygonField): 
291          symbolizer = PolygonSymbolizer 
292      else: 
293          # PointField, MultiPointField, GeometryField, or GeometryCollectionField 
294          symbolizer = PointSymbolizer 
295   
296      if propertyname is None: 
297          propertyname = field 
298   
299      datavalues = array(queryset.order_by(field).values_list(field, flat=True)) 
300      q = classification(datavalues, nclasses, **kwargs) 
301   
302      nl = thesld.create_namedlayer('%d breaks on "%s" as %s' % (nclasses, field, classification.__name__)) 
303      us = nl.create_userstyle() 
304      if not userstyletitle is None: 
305          us.Title = str(userstyletitle) 
306      fts = us.create_featuretypestyle() 
307      if not featuretypestylename is None: 
308          fts.Name = str(featuretypestylename) 
309   
310      for i,qbin in enumerate(q.bins): 
311          if type(qbin) == ndarray: 
312              qbin = qbin[0] 
313   
314          title = '<= %s' % qbin

316   
317          shade = float(i) / (q.k - 1.0) * 255 
318          if i == q.k:

322   
323          if symbolizer == PointSymbolizer:
